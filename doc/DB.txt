Пример запросов для аналитики
sql
Copy
-- Лидерборд за неделю
SELECT p.name, SUM(g.score) as weekly_score
FROM players p
JOIN games g ON p.id = g.player_id
WHERE g.started_at > datetime('now', '-7 days')
GROUP BY p.id
ORDER BY weekly_score DESC
LIMIT 100;

-- Статистика по часам (когда игрок активен)
SELECT hour_of_day, COUNT(*) as games_count
FROM games
WHERE player_id = 'player_001'
GROUP BY hour_of_day;

-- Винрейт по первой собранной масти
SELECT first_suit, 
       COUNT(*) as total,
       SUM(CASE WHEN result = 'won' THEN 1 ELSE 0 END) as wins,
       ROUND(100.0 * SUM(CASE WHEN result = 'won' THEN 1 ELSE 0 END) / COUNT(*), 2) as winrate
FROM games
WHERE player_id = 'player_001' AND first_suit IS NOT NULL
GROUP BY first_suit;

-- Прогресс по скрытым ачивкам (для подсказки "вы близки")
SELECT ad.name, pa.progress, ad.condition_data
FROM player_achievements pa
JOIN achievement_defs ad ON pa.achievement_id = ad.id
WHERE pa.player_id = 'player_001' 
  AND ad.is_hidden = 1 
  AND pa.unlocked_at IS NULL
  AND pa.progress > pa.target * 0.7;  -- больше 70%
Миграции при изменениях
Поскольку изменения редкие, простая система миграций:
Python
Copy
# migrations/001_initial.sql
# migrations/002_add_loose_streak.sql
# migrations/003_add_scoring.sql

def migrate(db_path):
    conn = sqlite3.connect(db_path)
    current_version = conn.execute("PRAGMA user_version").fetchone()[0]
    
    migrations = [
        '001_initial.sql',
        '002_add_loose_streak.sql', 
        '003_add_scoring.sql'
    ]
    
    for i, migration in enumerate(migrations[current_version:], start=current_version):
        with open(f'migrations/{migration}') as f:
            conn.executescript(f.read())
        conn.execute(f"PRAGMA user_version = {i + 1}")
        conn.commit()

# stats/event_bus.py
class EventBus:
    def __init__(self):
        self._handlers = {}
    
    def on(self, event_type, handler):
        self._handlers.setdefault(event_type, []).append(handler)
    
    def emit(self, event_type, data):
        for handler in self._handlers.get(event_type, []):
            handler(data)

# Глобальный инстанс
event_bus = EventBus()

gdscript
Copy
extends Node

const ICONS_PATH = "res://assets/achievements/icons/"
const BACKGROUNDS_PATH = "res://assets/achievements/backgrounds/"

# Кэш загруженных текстур
var _cache := {}

# Получить иконку с учётом состояния
func get_achievement_icon(achievement_id: String, status: Dictionary) -> Texture2D:
    var texture_path: String
    
    if status.get("hidden", false) and not status.get("unlocked", false):
        # Скрытая и не открыта — показываем вопрос
        texture_path = ICONS_PATH + "default_secret.png"
    elif not status.get("unlocked", false):
        # Не открыта — замок
        texture_path = ICONS_PATH + "default_locked.png"
    else:
        # Открыта — реальная иконка или fallback
        texture_path = ICONS_PATH + achievement_id + ".png"
        if not FileAccess.file_exists(texture_path):
            texture_path = ICONS_PATH + "default_locked.png"
    
    return _load_texture(texture_path)

func get_tier_background(tier: String) -> Texture2D:
    var path = BACKGROUNDS_PATH + "tier_" + tier + ".png"
    return _load_texture(path)

func _load_texture(path: String) -> Texture2D:
    if path in _cache:
        return _cache[path]
    
    var texture = load(path) as Texture2D
    if texture:
        _cache[path] = texture
    else:
        push_warning("Achievement texture not found: " + path)
        # Fallback на пустую текстуру или дефолт
    return texture

from PIL import Image, ImageDraw, ImageFont

TIERS = {
    'bronze': {'bg': '#CD7F32', 'border': '#8B4513'},
    'silver': {'bg': '#C0C0C0', 'border': '#808080'},
    'gold': {'bg': '#FFD700', 'border': '#B8860B'},
    'platinum': {'bg': '#E5E4E2', 'border': '#A0A0A0'},
}

def generate_icon(achievement_id, tier, symbol, output_path):
    size = 128
    img = Image.new('RGBA', (size, size), (0,0,0,0))
    draw = ImageDraw.Draw(img)
    
    # Фон (круг)
    margin = 4
    draw.ellipse([margin, margin, size-margin, size-margin], 
                 fill=TIERS[tier]['bg'], 
                 outline=TIERS[tier]['border'], 
                 width=4)
    
    # Символ (эмодзи или текст)
    font = ImageFont.truetype("arial.ttf", 64)
    bbox = draw.textbbox((0,0), symbol, font=font)
    text_w = bbox[2] - bbox[0]
    text_h = bbox[3] - bbox[1]
    x = (size - text_w) // 2
    y = (size - text_h) // 2 - 10
    draw.text((x, y), symbol, font=font, fill='white')
    
    img.save(output_path)

from flask import Flask, jsonify
from .repository import AchievementRepository

@app.route('/api/achievements/<player_id>')
def get_achievements(player_id):
    repo = AchievementRepository()
    
    # Получаем все определения ачивок
    defs = repo.get_all_definitions()
    
    # Получаем прогресс игрока
    progress = repo.get_player_progress(player_id)
    
    result = []
    for d in defs:
        p = progress.get(d['id'], {
            'unlocked': False,
            'progress': 0,
            'target': d['condition_data'].get('threshold', 1)
        })
        
        result.append({
            'id': d['id'],
            'data': {
                'name': d['name'],
                'description': d['description'] if not d['is_hidden'] or p['unlocked'] else None,
                'category': d['category'],
                'tier': _get_tier(d),  # bronze/silver/gold по сложности
                'hidden': d['is_hidden']
            },
            'status': {
                'unlocked': p['unlocked'],
                'unlocked_at': p.get('unlocked_at'),
                'progress': p['progress'],
                'target': p['target'],
                'hidden': d['is_hidden'] and not p['unlocked']
            }
        })
    
    return jsonify({'achievements': result})

def _get_tier(definition):
    # Логика определения тира по сложности
    if '1000' in str(definition['condition_data']):
        return 'platinum'
    elif '100' in str(definition['condition_data']):
        return 'gold'
    elif '10' in str(definition['condition_data']):
        return 'silver'
    return 'bronze'

Выбор способа идентификации игрока
Вариант 1: Локальный UUID (без авторизации)
Генерация uuid4() при первом запуске Godot
Хранение в user://player_id.save
Передача в HTTP-заголовке или теле запроса
Плюс: максимальная простота, нет входа
Минус: подмена ключа, потеря прогресса при смене устройства
Вариант 2: UUID + простая защита от подмены
Тот же UUID, но + подпись запросов
Сервер выдаёт секрет при первом обращении
Клиент подписывает запросы HMAC-SHA256
Плюс: защита от простой подмены ID
Минус: сложнее реализация, всё ещё нет переноса прогресса
Вариант 3: Google OAuth 2.0 (рекомендую)
Godot открывает браузер/вебвью для авторизации
Google возвращает id_token (JWT)
Сервер верифицирует токен, извлекает sub (стабильный ID пользователя)
Плюс: реальный аккаунт, перенос между устройствами, защищённость
Минус: требует интернета, настройка в Google Cloud Console
Вариант 4: Гибрид (UUID + опциональная привязка Google)
Старт с локальным UUID, анонимная статистика
Предложение привязать Google при достижении N побед
Миграция прогресса при привязке
Плюс: нет барьера входа, сохранность данных потом
Минус: сложная логика слияния прогресса
Сравнение для вашего случая

# stats/data/schema.py
import sqlite3
import os
from pathlib import Path


def get_db_path() -> str:
    """Возвращает путь к файлу базы данных."""
    # База данных в папке data рядом со скриптом
    current_dir = Path(__file__).parent
    db_path = current_dir / "patience.db"
    return str(db_path)


def create_tables(conn: sqlite3.Connection) -> None:
    """Создаёт таблицы базы данных."""
    
    cursor = conn.cursor()
    
    # Таблица игроков
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS players (
            id TEXT PRIMARY KEY,
            name TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            last_played TIMESTAMP,
            
            -- Базовая статистика
            games_started INTEGER DEFAULT 0,
            games_won INTEGER DEFAULT 0,
            games_lost INTEGER DEFAULT 0,
            games_abandoned INTEGER DEFAULT 0,
            
            -- Серии
            current_win_streak INTEGER DEFAULT 0,
            best_win_streak INTEGER DEFAULT 0,
            current_loose_streak INTEGER DEFAULT 0,
            best_loose_streak INTEGER DEFAULT 0,
            
            -- Очки
            total_score INTEGER DEFAULT 0,
            highest_score INTEGER DEFAULT 0,
            
            -- Время
            total_play_time_seconds INTEGER DEFAULT 0,
            fastest_win_seconds INTEGER,
            slowest_win_seconds INTEGER,
            
            version INTEGER DEFAULT 1
        )
    """)
    
    # Таблица игр
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS games (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            player_id TEXT NOT NULL REFERENCES players(id),
            
            -- Время
            started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            ended_at TIMESTAMP,
            
            -- Результат
            result TEXT CHECK(result IN ('won', 'lost', 'abandoned')),
            
            -- Метрики
            score INTEGER,
            duration_seconds INTEGER,
            moves_count INTEGER DEFAULT 0,
            undos_used INTEGER DEFAULT 0,
            hints_used INTEGER DEFAULT 0,
            deck_cycles INTEGER DEFAULT 0,
            
            -- Дополнительно для аналитики
            suits_completed TEXT,  -- JSON массив
            first_suit TEXT,
            was_perfect BOOLEAN DEFAULT 0,
            
            hour_of_day INTEGER,
            day_of_week INTEGER,
            is_weekend BOOLEAN
        )
    """)
    
    # Индексы для производительности
    cursor.execute("""
        CREATE INDEX IF NOT EXISTS idx_games_player 
        ON games(player_id)
    """)
    
    cursor.execute("""
        CREATE INDEX IF NOT EXISTS idx_games_player_date 
        ON games(player_id, started_at)
    """)
    
    conn.commit()


def init_database() -> None:
    """Инициализирует базу данных, создаёт если не существует."""
    db_path = get_db_path()
    
    # Создаём директорию если нужно
    os.makedirs(os.path.dirname(db_path), exist_ok=True)
    
    conn = sqlite3.connect(db_path)
    try:
        create_tables(conn)
        print(f"Database initialized: {db_path}")
    finally:
        conn.close()


if __name__ == "__main__":
    init_database()
